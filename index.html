    <h2>CPM Calculator</h2>

    <input type="number" id="cpm" placeholder="Enter CPM ($)" />

    <input type="number" id="ads" placeholder="Enter Number of Ads" />

    <input type="number" id="listeners" placeholder="Enter Monthly Listeners" />

    <button onclick="calculateRevenue()">Calculate Revenue</button>

    <div id="result">Result will appear here</div>

</div>


<script>

    function calculateRevenue() {

        var cpm = document.getElementById('cpm').value;

        var ads = document.getElementById('ads').value;

        var listeners = document.getElementById('listeners').value;


        if (cpm && ads && listeners) {

            var monthlyRevenue = (listeners / 1000) * cpm * ads;

            document.getElementById('result').innerText = 'Monthly Revenue: $' + monthlyRevenue.toFixed(2);

        } else {

            document.getElementById('result').innerText = 'Please fill in all fields.';

        }

    }

</script>
