    <input type="number" id="listeners" placeholder="Enter Monthly Listeners" />

    <button onclick="calculateRevenue()">Calculate Revenue</button>

<script>

    function calculateRevenue() {

        const cpm = document.getElementById('cpm').value;

        const ads = document.getElementById('ads').value;

        const listeners = document.getElementById('listeners').value;


        if (cpm && ads && listeners) {

            const monthlyRevenue = (listeners / 1000) * cpm * ads;

            const annualRevenue = monthlyRevenue * 12;


            // Display the results in the "result" div

            document.getElementById('result').innerHTML = `

                Monthly Revenue: $${monthlyRevenue.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2})}<br>

                Annual Revenue: $${annualRevenue.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2})}

            `;

        } else {

            document.getElementById('result').innerHTML = 'Please fill in all fields.';

        }

    }

</script>
